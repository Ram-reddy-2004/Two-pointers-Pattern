class Solution {
    public int[] twoSum(int[] numbers, int target) {
        int left=0, right = numbers.length -1;
        int[] res = new int[2]; 
        int sum;
        while(left < right){
            sum = numbers[left] + numbers[right];
            if(target == sum){
                res[0] = left+1;
                res[1] = right+1;
                break;
            }
            else if( sum < target ) left++;
            else right--;
        }
        return res;
    }
}
